<script>
import WindowPortal from "@/components/WindowPortal.vue";
import Graphs from "@/components/Graphs/Graphs.vue";


export default {
    name: "Capteur_elec",
    components: {Graphs, WindowPortal},
    props: {
        name: String,
        from: String,
        to: String
    },
    data() {
        return {
            open : false
        }
    },
    methods:{
        click(){
            if (this.open === false)
                this.open = true
            else
                this.open = false
        }
    }
}
</script>

<template>
    <button class="btn btn-primary" @click="click"/>
    <WindowPortal :open="open">
        <div class="text-center">
            <Graphs :name="name" :from="from" :to="to"/>
        </div>
    </WindowPortal>
</template>
